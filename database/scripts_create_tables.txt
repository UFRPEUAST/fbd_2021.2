CREATE TABLE IF NOT EXISTS Curso (
    id serial PRIMARY key,
    nome varchar(255),
    sigla varchar(10),
    status bool default TRUE);

CREATE TABLE IF NOT EXISTS Grade(
    id serial PRIMARY key,
    titulo     varchar(255),
    carga_horaria float,
    status bool default TRUE,
    duracao int default 0,
    id_curso int REFERENCES Curso(id));
